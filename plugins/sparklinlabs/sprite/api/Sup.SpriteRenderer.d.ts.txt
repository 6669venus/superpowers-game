declare module Sup {
  class SpriteRenderer extends ActorComponent {
    constructor(actor: Actor, asset?: Sprite, materialType?: SpriteRenderer.MaterialType );

    getSprite(): Sprite;
    setSprite(asset: Sprite, materialType?: SpriteRenderer.MaterialType): SpriteRenderer;
    getMaterialType(): SpriteRenderer.MaterialType;
    getOpacity(): number;
    setOpacity(opacity: number): SpriteRenderer;
    getColor(): Sup.Color;
    setColor(color: Sup.Color): SpriteRenderer;

    getAnimation(): string;
    setAnimation(animationName: string, looping?: boolean): SpriteRenderer;
    setAnimationTime(time: number): SpriteRenderer;
    getAnimationTime(): number;
    getAnimationDuration(): number;

    isAnimationPlaying(): boolean;
    playAnimation(looping?: boolean): SpriteRenderer;
    pauseAnimation(): SpriteRenderer;
    stopAnimation(): SpriteRenderer;
  }

  module SpriteRenderer {
    enum MaterialType { Basic, Phong }
  }
}

declare module Sup {

  class TextRenderer extends ActorComponent {
    constructor(actor: Actor, text?: string, asset?: Font, options?: {alignment?: string; size?: number; color?: string;});

    getFont(): Font;
    setFont(font: Font): TextRenderer;
    getText(): string;
    setText(text: string): TextRenderer;
    getColor(): string;
    setColor(color: string): TextRenderer;
    getAlignment(): string;
    setAlignment(alignment: string): TextRenderer;
    getSize(): number;
    setSize(size: number): TextRenderer;
    getOpacity(): number;
    setOpacity(opacity: number): TextRenderer;
  }
}

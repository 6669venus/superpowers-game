declare module Sup {
  module Audio {
    class SoundInstance {
      constructor(pathOrAsset: string|Sound);
      play(): SoundInstance;
      stop(): SoundInstance;
      pause(): SoundInstance;

      getLoop(): boolean;
      setLoop(looping: boolean): SoundInstance;
      getVolume(): number;
      setVolume(volume: number): SoundInstance;
      getPan(): number;
      setPan(pan: number): SoundInstance;
      getPitch(): number;
      setPitch(pitch: number): SoundInstance;
    }
  }
}
